45 juce::AudioThumbnail &getOrCreateThumbnail(juce::ARAAudioSource *audioSource,
46 AudioModification *audioModification) {
47 const auto iter = thumbnails.find(audioSource);
48
49 // if a thumbnail was found for this source,
50 if (iter != std::end(thumbnails))
51 if (audioModification->getIsModified()){
52 // if the audio has been modified even once,
53 // this flag will remain true forever.
54 // That's why we'll check if a thumbnail has been
55 // created for the latest processing step.
56 if (audioModification->isThumbCreated()){
57 // it means the current thumbnail is the updated one,
58 // and we just need to return it.
59 return *iter->second;
60 }
61 else{
62 // It means there has been a processing step since
63 // the last time the thumbnail was created, so we need
64 // to update it.
65 ++hash;
66 iter->second->setSource(audioModification->getModifiedAudioBuffer(),
67 audioSource->getSampleRate(), hash);
68 audioModification->setThumbCreated(true);
69 return *iter->second;
70 }
71 }
72 else{
73 // if the source has never been modified then
74 // we just return the thumbnail previously created.
75 return *iter->second;
76 }
77
78
79 // if not found, create the thumbnail
80 auto thumb =
81 std::make_unique<juce::AudioThumbnail>(128, dummyManager, thumbnailCache);
82 auto &result = *thumb;
83
84 ++hash;
85 if (audioModification->getIsModified())
86 thumb->setSource(audioModification->getModifiedAudioBuffer(),
87 audioSource->getSampleRate(), hash);
88
89 else
90 thumb->setReader(new juce::ARAAudioSourceReader(audioSource), hash);
91
92 audioSource->addListener(this);
93 thumbnails.emplace(audioSource, std::move(thumb));
94 audioModification->setThumbCreated(true);
95 return result;
96 }
