29 Looper(const juce::AudioBuffer<float> *buffer, juce::Range<juce::int64> range)

32 void writeInto(juce::AudioBuffer<float> &buffer) {
33 if (loopRange.getLength() == 0) {
34 buffer.clear();
35 return;
36 }
37
38 const auto numChannelsToCopy =
39 std::min(inputBuffer->getNumChannels(), buffer.getNumChannels());
40 const auto actualCrossfadeLengthSamples = std::min(
41 loopRange.getLength() / 2, (juce::int64)desiredCrossfadeLengthSamples);
42
43 for (auto samplesCopied = 0; samplesCopied < buffer.getNumSamples();) {
44 const auto [needsCrossfade, samplePosOfNextCrossfadeTransition] =
45 [&]() -> std::pair<bool, juce::int64> {
46 if (const auto endOfFadeIn =
47 loopRange.getStart() + actualCrossfadeLengthSamples;
48 pos < endOfFadeIn)
49 return {true, endOfFadeIn};
50
51 return {false, loopRange.getEnd() - actualCrossfadeLengthSamples};
52 }();
53
54 const auto samplesToNextCrossfadeTransition =
55 samplePosOfNextCrossfadeTransition - pos;
56 const auto numSamplesToCopy =
57 std::min(buffer.getNumSamples() - samplesCopied,
58 (int)samplesToNextCrossfadeTransition);
59
60 const auto getFadeInGainAtPos = [this,
61 actualCrossfadeLengthSamples](auto p) {
62 return juce::jmap((float)p, (float)loopRange.getStart(),
63 (float)loopRange.getStart() +
64 (float)actualCrossfadeLengthSamples - 1.0f,
65 0.0f, 1.0f);
66 };
67
68 for (int i = 0; i < numChannelsToCopy; ++i) {
69 if (needsCrossfade) {
70 const auto overlapStart = loopRange.getEnd() -
71 actualCrossfadeLengthSamples +
72 (pos - loopRange.getStart());
73
74 crossfade(inputBuffer->getReadPointer(i, (int)pos),
75 inputBuffer->getReadPointer(i, (int)overlapStart),
76 getFadeInGainAtPos(pos),
77 getFadeInGainAtPos(pos + numSamplesToCopy),
78 buffer.getWritePointer(i, samplesCopied), numSamplesToCopy);
79 } else {
80 buffer.copyFrom(i, samplesCopied, *inputBuffer, i, (int)pos,
81 numSamplesToCopy);
82 }
83 }
84
85 samplesCopied += numSamplesToCopy;
86 pos += numSamplesToCopy;
87
88 jassert(pos <= loopRange.getEnd() - actualCrossfadeLengthSamples);
89
90 if (pos == loopRange.getEnd() - actualCrossfadeLengthSamples)
91 pos = loopRange.getStart();
92 }
93 }
94
95private:
96 static constexpr int desiredCrossfadeLengthSamples = 50;
